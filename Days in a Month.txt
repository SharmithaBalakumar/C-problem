#include <stdio.h>
int main(){
    int number;
    scanf("%d",&number);
    switch(number){
        case 1:
        printf("January has 31 days");
        break;
        case 2:
        printf("February has 28 or 29 days");
        break;
        case 3:
        printf("March has 31 days");
        break;
        case 4:
        printf("April has 30 days");
        break;
        default:
        printf("Invalid month");
        
        
    }
    return 0;
}
