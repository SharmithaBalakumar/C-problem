#include <stdio.h>
int main(){
    char character;
    scanf("%c",&character);
    if(character>=0&&character<=9){
        printf("Digit");
    }
    else if(character>='A'&&character<='Z'||character>='a'&&character<='z'){
        printf("Alphapet");
    }
    else{
        printf("Special character");
    }
    return 0;
}
