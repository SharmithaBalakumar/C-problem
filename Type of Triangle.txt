#include <stdio.h>
int main(){
    int side1;
    int side2;
    int side3;
    scanf("%d",&side1);
    scanf("%d",&side2);
    scanf("%d",&side3);
    if(side1==side2 && side2==side3 && side3==side1){
        printf("Equilateral triangle");
    }
    else{
        printf("Not a equilateral triangle");
    }
}
